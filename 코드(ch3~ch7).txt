#include <stdio.h>
int main(void)
{
	int first, second;
	int result;

	printf("두 개의 정수를 입력(1st<2nd): ");
	scanf("%d %d", &first, &second);

	for (result = 0; first <= second; first++)
		result += first;

	printf("결과값 : %d", result);
	
	return 0;
}

OR

#include <stdio.h>

int total(int n1, int n2);

int main()
{
	int num1, num2;

	printf("두 개의 정수를 입력 : ");
	scanf("%d %d", &num1, &num2);

	if (num1 > num2)
		printf("%d", total(num2, num1));
	else
		printf("%d", total(num1, num2));

	return 0;
}

int total(int n1, int n2)
{
	int sum = 0;
	for (int i = n1; i <= n2; i++)
		sum += i;
	return sum;
}
